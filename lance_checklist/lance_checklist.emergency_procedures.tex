\phantomsection
\addcontentsline{toc}{section}{Emergency Procedures}

\twocolumn

\begin{checklist_emerg}{Engine Fire During Start}
    \checkitem{Starter}{crank engine}
    \checkitem{Mixture}{idle cut-off}
    \checkitem{Throttle}{open}
    \checkitem{Fuel pump}{off}
    \checkitem{Fuel selector}{off}
    Abandon if fire continues.
\end{checklist_emerg}

\begin{checklist_emerg}{Engine Power Loss In Flight}
    \checkitem{Airspeed}{best glide, 92 KIAS}
    \checkitem{Landing site}{pick}
    \begin{center}
        \textbf{If sufficient altitude:}
    \end{center}
    \checkitem{Fuel selector}{switch}
    \checkitem{Fuel pump}{on}
    \checkitem{Mixture}{full rich}
    \checkitem{Alternate air}{open}
    \checkitem{Engine gauges}{check}
\end{checklist_emerg}

\begin{checklist_emerg}{Electrical Fire}
    \checkitem{Master switch}{off}
    \checkitem{Avionics master}{off}
    \checkitem{Electrical switches}{off}
    \begin{center}
        \textbf{If no smoke:}
    \end{center}
    \checkitem{Circuit breakers}{note tripped}
    \checkitem{Circuit breakers}{off}
    \checkitem{Master switch}{on}
    \begin{center}
        \textbf{If no smoke:}
    \end{center}
    \checkitem{Avionics master}{on}
\end{checklist_emerg}

\begin{checklist_emerg}{Engine Fire}
    \checkitem{Fuel selector}{off}
    \checkitem{Throttle}{closed}
    \checkitem{Mixture}{idle cut-off}
    \checkitem{Fuel pump}{off}
    \checkitem{Heater and defroster}{closed}
\end{checklist_emerg}

\begin{checklist_emerg}{Alternator Failure}
    Verify failure. 
    \\
    Reduce electrical load as much as possible.
    \\
    \checkitem{Alt circuit breakers}{check}
    \checkitem{Alt switch}{off, wait, then on}
    \begin{center}
        \textbf{If no output:}
    \end{center}
    \checkitem{Alt switch}{off}
\end{checklist_emerg}

\begin{checklist_emerg}{Propeller Overspeed}
    \checkitem{Throttle}{retard}
    \checkitem{Oil pressure}{check}
    \checkitem{Prop}{decrease}
    \checkitem{Airspeed}{reduce}
    \checkitem{Throttle}{below 2700 RPM}
\end{checklist_emerg}

\begin{checklist_emerg}{Emergency Landing Gear Extension}
    \checkitem{Master switch}{on}
    \checkitem{Circuit breakers}{check}
    \checkitem{Dome lights}{off}
    \begin{center}
        \textbf{If landing gear does not check down and locked:}
    \end{center}
    \checkitem{Airspeed}{below 87 KIAS}
    \checkitem{Landing gear selector}{down}
    \begin{center}
        \textbf{If landing gear still does not check down and locked:}
    \end{center}
    \checkitem{Emergency gear lever}{down}
    \begin{center}
        (while fishtailing airplane)
    \end{center}
\end{checklist_emerg}

\begin{checklist_emerg}{IFR Comms Failure}
    \checkitem{Squwak}{7600}

    \begin{center}
        \textbf{If VFR, land as soon as practicable.}
    \end{center}

    \textbf{Route}
    \begin{itemize}[noitemsep]
        \item ATC assigned Route
        \item Vector clearance
        \item Expected route
        \item Filed route
    \end{itemize}

    \textbf{Altitude (highest of)}
    \begin{itemize}[noitemsep]
        \item Minimum altitude for IFR
        \item Expected altitude
        \item Assigned altitude
    \end{itemize}
\end{checklist_emerg}
